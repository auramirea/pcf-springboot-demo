Tips

Use 

management.security.enabled=false
in your 
application.properties
to turn off the management API security.

Properties

Set these in your 

application.properties
for more interesting output in the 
/info
endpoint...

info.os = ${os.name}
info.port = ${local.server.port}
info.javaName = ${java.runtime.name}
info.javaVmName = ${java.vm.name}
info.javaVersion = ${java.runtime.version}
info.profiles = ${profiles}
info.applicationIndex = ${INSTANCE_INDEX}
info.managementSecurityEnabled = ${management.security.enabled}
info.services-messaging = ${vcap.services.rabbit.name}
info.services-database = ${vcap.services.mysql.name}
info.applicationName = ${vcap.application.name}
info.applicationSpace = ${vcap.application.space_name}
info.applicationUri = ${vcap.application.uris}
info.applicationMemoryLimit = ${vcap.application.limits.mem}
info.applicationStorageLimit = ${vcap.application.limits.disk}
